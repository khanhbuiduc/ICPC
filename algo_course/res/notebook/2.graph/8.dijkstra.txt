typedef pair<int, int> pii;
priority_queue<pii, vector<pii>, greater<pii>> pq;
int dist[100005];

void dijkstra(int start)
{
    pq.emplace(0, start);
    while (!pq.empty())
    {
        auto [d, u] = pq.top();
        pq.pop();
        if (d > dist[u])
            continue;
        for (auto [v, w] : adj[u])
            if (dist[v] > dist[u] + w)
            {
                dist[v] = dist[u] + w;
                pq.emplace(dist[v], v);
            }
    }
}
