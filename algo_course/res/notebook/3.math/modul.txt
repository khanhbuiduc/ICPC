// Phép cộng
(a + b) % M = (a % M + b % M) % M

// Phép trừ
(a - b) % M = (a % M - b % M + M) % M

// Phép nhân
(a × b) % M = ([a % M] × [b % M]) % M

// Phép chia (với M là số nguyên tố)
1/a % M = a^(M-2) % M

// Phép mũ: sử dụng hàm powMod

ll powMod(ll a, ll b) {
    ll res = 1;
    while (b > 0) {
        if (b & 1)
            (res *= a) %= MOD;
        (a *= a) %= MOD;
        b >>= 1;
    }
    return res;
}
